## lineparine orthography

vowel = "a" | "i" | "u" | "e" | "o" | "y"
"a" -> /a/; "i" -> /i/; "u" -> /u/; "e" -> /e/; "o" -> /o/; "y" -> /y/

"p" -> /p/; "b" -> /b/; "t" -> /t/; "d" -> /d/; "k" -> /k/; "g" -> /g/;
"m" -> /m/; "n" -> /n/; "f" -> /f/; "v" -> /v/; 

"c" -> /s/; "z" -> /t͡s/;
"s" -> /z/ 
"x" -> /ʃ/
"h" -> /h/
"w" -> /w/
"j" -> /j/
"l" -> /l/
"q" -> /kw/
!vowel ("r" | "R") -> /r/
"ts" -> /t͡s/
"ch" -> /t͡ʃ/
"Z" -> /d͡ʑ/
"ng" -> /ŋ/
"F"  -> /ɸ/
"V"  -> /β/
"sh" -> /ʃ/
"th" -> /θ/
"dh" -> /ð/
"kh" -> /x/
!vowel ("rkh" | "Rkh") -> /ʁ/
"rl" -> /ɹ/
"Rl" -> /ɹ/

"i" vowel -> /j/ $2
"y" vowel -> /ɥ/ $2
vowel "y" !vowel -> $1 /ɥ/
vowel "r" -> $1 /ː/
"j" !vowel -> /i/
vowel "i" !"r" -> $1 /j/
vowel "u" !"r" -> $1 /w/
palat "i" "u" -> $1 $2 /u/

"ph" -> /p/
"s" !vowel -> /s/
# "l" !vowel -> /ʟ/

"'" -> //
palat = "Z" | "x" | "ch" | "sh"
palat "i" vowel -> $1 // $3

"tz" -> /t͡st͡s/;
"sx" -> /ʃʃ/;
"dZ" -> /d͡ʑd͡ʑ/;
"en" ^ -> /ən/;
"ni" vowel -> /ɲ/ $2;
"nj" vowel -> /ɲ/ $2;
"s" "j" !vowel -> /z/ $2;

`edioll` -> /edjol/
`wioll` -> /wjol/
`woll` -> /wol/
"vsfafgh" -> /vusfafuguh/
"simfgh" -> /zimfuguh/
